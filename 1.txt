        var foot = $(".foot").position().top;
        var heightLeft = $(".left").outerHeight();
        var heightRight = $(".right").outerHeight();
        var heightFoot = $(".foot").outerHeight();
        if (scroll == main || scroll > main) {
          $(".left").css({ "position": "fixed", "top": "0", "bottom": "auto" });
          $(".right").css({ "position": "fixed", "top": "0", "bottom": "auto" });
        }
        else {
          $(".left").css({ "position": "absolute", "top": "90px", "bottom": "auto" });
          $(".right").css({ "position": "absolute", "top": "90px", "bottom": "auto" });
        };
        if (scroll + heightLeft > foot || scroll + heightLeft == foot) {
          $(".left").css({ "position": "absolute", "top": "auto", "bottom": heightFoot + "px" });
          $(".right").css({ "position": "absolute", "top": "auto", "bottom": heightFoot + "px" });
        }




        $(".left h2").text(scroll);
        $(".left h3").text(left);
        $(".left h4").text(main);
        $(".left h5").text(foot);
      });


    });
